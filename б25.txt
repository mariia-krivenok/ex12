import tkinter as tk
from tkinter import messagebox


def calculate_trip_cost():
    try:
        # Получаем данные из полей
        daily_cost = float(daily_cost_entry.get())
        days = int(days_entry.get())
        visa_cost = float(visa_cost_entry.get())
        transport_cost = float(transport_cost_entry.get())

        # Расчет стоимости поездки
        base_cost = (daily_cost * days) + visa_cost + transport_cost
        total_cost = base_cost * 1.18  # Добавляем 18% НДС

        # Отображаем результат
        result_label.config(text=f"Итоговая стоимость поездки: {total_cost:.2f} руб.")
    except ValueError:
        messagebox.showerror("Ошибка", "Пожалуйста, введите корректные числовые значения!")


# Создание основного окна
root = tk.Tk()
root.title("Расчет стоимости поездки")

# Поля ввода данных
tk.Label(root, text="Стоимость 1 дня пребывания (руб.):").pack(pady=5)
daily_cost_entry = tk.Entry(root)
daily_cost_entry.pack(pady=5)

tk.Label(root, text="Количество дней пребывания:").pack(pady=5)
days_entry = tk.Entry(root)
days_entry.pack(pady=5)

tk.Label(root, text="Стоимость оформления визы (руб.):").pack(pady=5)
visa_cost_entry = tk.Entry(root)
visa_cost_entry.pack(pady=5)

tk.Label(root, text="Стоимость транспортных услуг (руб.):").pack(pady=5)
transport_cost_entry = tk.Entry(root)
transport_cost_entry.pack(pady=5)

# Кнопка для расчета
calculate_button = tk.Button(root, text="Рассчитать стоимость", command=calculate_trip_cost)
calculate_button.pack(pady=10)

# Метка для отображения результата
result_label = tk.Label(root, text="", font=("Arial", 12))
result_label.pack(pady=10)

# Запуск основного цикла приложения
root.mainloop()
